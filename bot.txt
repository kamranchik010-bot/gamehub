    <script>
        const signalOutput = document.getElementById('signal_output');

        function getSignal(game) {
            signalOutput.innerHTML = `⏳ **${game.toUpperCase()}** o'yini uchun signal tayyorlanmoqda...`;
            
            // Hozircha bu yerda faqatgina tasodifiy (random) signal yaratamiz
            setTimeout(() => {
                let signal;
                if (game === 'mines') {
                    // MINES: 25 ta katak ichidan 5 ta xavfsiz katakni tasodifiy tanlash
                    const safe_cells = Array.from({length: 25}, (_, i) => i + 1)
                                            .sort(() => 0.5 - Math.random())
                                            .slice(0, 5)
                                            .join(', ');
                    signal = `✅ Xavfsiz kataklar: ${safe_cells}. Avto-tikishga o'ting!`;

                } else if (game === 'roulette') {
                    // ROULETTE: Qizil yoki Qora
                    const colors = ['🔴 Qizil', '⚫ Qora'];
                    const chosen_color = colors[Math.floor(Math.random() * colors.length)];
                    signal = `🔥 Hozirgi signal: **${chosen_color}** ga tiking!`;
                    
                } else if (game === 'gems') {
                    // GEMS: 3 ta katakdan 2 ta olmos (masalan, 1-katak, 3-katak olmos)
                    const gems_cells = Array.from({length: 3}, (_, i) => i + 1)
                                            .sort(() => 0.5 - Math.random())
                                            .slice(0, 2)
                                            .join(' va ');
                    signal = `💎 Olmoslar joylashuvi: **${gems_cells}**-katak. Uchinchi katak bo'sh.`;
                }

                signalOutput.innerHTML = signal;
            }, 1500); // 1.5 soniyadan keyin signalni ko'rsatish
        }
    </script>
